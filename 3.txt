create TABLE suppliers(
  	sid int,
    sname varchar(20),
    address varchar(40),
    PRIMARY key (sid)
)

create TABLE parts(
  	pid int,
    pname varchar(20),
    color varchar(40),
    PRIMARY key (pid)
)

create TABLE catalog(
  	sid int,
  	pid int,
    cost REAL,
    FOREIGN key (sid) references suppliers(sid),
    FOREIGN key (pid) references parts(pid)
)
