create TABLE flights(
    flno int,
    frm VARCHAR(20),
    destination VARCHAR(20),
    distance INT,
    departs TIME,
    arrives time,
    price INT,
    primary key(flno)
);

create TABLE aircraft(
    aid int,
    aname VARCHAR(5),
    cruisingRange int,
    PRIMARY key (aid)
);

create TABLE certified(
    eid int,
    aid INT,
    FOREIGN key (eid) REFERENCES flights(eid),
    FOREIGN key (aid) REFERENCES aircraft(aid),
);

create TABLE employee(
    eid int,
    ename VARCHAR,
    salary int,
    primary key (eid)
);
